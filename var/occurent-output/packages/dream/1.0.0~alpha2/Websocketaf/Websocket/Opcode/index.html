type standard_non_control = [
| `Continuation
| `Text
| `Binary
]
type standard_control = [
| `Connection_close
| `Ping
| `Pong
]
type standard = [
| standard_non_control
| standard_control
]
type t = [
| standard
| `Other of int
]
val code : t -> int
val of_code : int -> t option
val of_code_exn : int -> t
val to_int : t -> int
val of_int : int -> t option
val of_int_exn : int -> t
val pp_hum : Format.formatter -> t -> unit
